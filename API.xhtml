caworks Server Api

 

What is the API?
You can use an SL script query different data.
For example:
* uuid of an avatar
* Texturekey of a sim
* all the important information of a gate from the gate network

 

To retrieve an uuid from an avatar:

http://cyber.caworks-sl.de/API/api.php?key='your_api_key'&action=n2k&var=cyber%20Ahn

---------------------------------------------------------

To get a Texturekey of a Sim:

http://cyber.caworks-sl.de/API/api.php?key='your_api_key'&action=simmap&var=Lychee%20Island

----------------------------------------------------------

To retrieve information about a gate:

http://cyber.caworks-sl.de/API/api.php?key='your_api_key'&action=gatesearch&var=kxworks

 

on position kxworks you can use:

GateName,SimName or Owner of the Gate

 

Returned Data:

GateName|Sim|Position|Owner|Model|Address

----------------------------------------------------------------

To post a message to an object

http://cyber.caworks-sl.de/API/api.php?key='your_api_key'&action=comm&var=your-message&chan=1&obid=http-uuid

 

For example scripts see below.

To get an API Key to write an IM inworld to cyber81 Bellic.

*****************************************************************************

Example LSL code for NameToKey:

key reqid;

default

{

         state_entry()

    {

    }

            touch_start(integer total_number)

    {

                reqid = llHTTPRequest("http://cyber.caworks-sl.de/API/api.php?key=your api key&action=n2k&var="+llEscapeURL("cyber Ahn"), [], "");

    }

            http_response(key id, integer status, list meta, string body)

    {

                        if ( id == reqid )

        {

                            llSay(0,body);

        }

    }

}

receiver

----------

 

key reqid;

default

{

    state_entry()

    {

        llOpenRemoteDataChannel();

    }

    remote_data(integer type, key channel, key message_id, string sender, integer ival, string sval)

    {

        if (type == REMOTE_DATA_CHANNEL)

        {

            string id_http = channel;

            llOwnerSay( "http_uuid = "+ id_http); //uuid for the transmitter

        }

        if (type == REMOTE_DATA_REQUEST)

        {

            llSay(0,"Message = "+sval+" and Channel = "+(string)ival);

        }

    }

}
